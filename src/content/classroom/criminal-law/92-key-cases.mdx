These opinions supply the doctrinal touchstones for analyzing liability, defenses, and punishment in criminal law.

<CaseCard
  title="Morissette v. United States (1952)"
  holding="A conviction for converting government property required proof of criminal intent because Congress did not clearly dispense with mens rea."
  rule="Courts presume a mental state requirement for traditional crimes unless the statute unmistakably creates strict liability."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/342/246",
    oyez: "https://www.oyez.org/cases/1940-1955/342us246"
  }}
/>

<CaseCard
  title="Mullaney v. Wilbur (1975)"
  holding="Maine violated due process by shifting to the defendant the burden of proving he acted in the heat of passion to reduce murder to manslaughter."
  rule="The prosecution must prove beyond a reasonable doubt every fact necessary to constitute the crime, including the absence of heat of passion when it is part of the statutory offense."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/421/684",
    oyez: "https://www.oyez.org/cases/1974/73-1302"
  }}
/>

<CaseCard
  title="Patterson v. New York (1977)"
  holding="New York could require defendants to prove the affirmative defense of extreme emotional disturbance without violating due process."
  rule="States may place the burden of persuasion on defendants for true affirmative defenses so long as the prosecution still proves the statutory elements beyond a reasonable doubt."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/432/197",
    oyez: "https://www.oyez.org/cases/1976/75-1057"
  }}
/>

<CaseCard
  title="Roper v. Simmons (2005)"
  holding="Executing offenders who were under 18 at the time of their crimes violates the Eighth Amendment."
  rule="The Constitution bars capital punishment for juveniles, reflecting evolving standards of decency and proportionality limits on punishment."
  links={{
    lii: "https://www.law.cornell.edu/supct/html/03-633.ZS.html",
    oyez: "https://www.oyez.org/cases/2004/03-633"
  }}
