These cornerstone cases ground the rules that police, prosecutors, and defense lawyers navigate every day.

<CaseCard
  title="Katz v. United States (1967)"
  holding="Electronic surveillance of a phone booth without a warrant violated the Fourth Amendment because the defendant expected privacy in his conversation."
  rule="The Fourth Amendment protects people, not places, guarding reasonable expectations of privacy and requiring warrants absent a recognized exception."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/389/347",
    oyez: "https://www.oyez.org/cases/1967/35"
  }}
/>

<CaseCard
  title="Terry v. Ohio (1968)"
  holding="An officer may briefly detain and frisk a suspect based on reasonable suspicion that criminal activity is afoot and the person is armed."
  rule="Stop-and-frisk encounters require articulable, specific facts supporting suspicion; the frisk must be limited to a protective pat-down for weapons."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/392/1",
    oyez: "https://www.oyez.org/cases/1967/67"
  }}
/>

<CaseCard
  title="Miranda v. Arizona (1966)"
  holding="Statements from custodial interrogation are inadmissible unless police advise suspects of their rights to remain silent and to counsel."
  rule="Custodial questioning triggers procedural safeguards that require warnings and a voluntary, knowing waiver before statements may be used in the prosecution's case-in-chief."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/384/436",
    oyez: "https://www.oyez.org/cases/1965/759"
  }}
/>

<CaseCard
  title="Gideon v. Wainwright (1963)"
  holding="The Sixth Amendment requires states to provide counsel to indigent defendants charged with serious crimes."
  rule="In felony prosecutions, the right to counsel is fundamental and binding on the states through the Fourteenth Amendment, obligating appointment of attorneys for defendants who cannot afford one."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/372/335",
    oyez: "https://www.oyez.org/cases/1962/155"
  }}
/>
