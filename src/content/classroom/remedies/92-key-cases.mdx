Keep these precedents in mind when deciding what relief to seek and how to defend against it.

<CaseCard
  title="eBay Inc. v. MercExchange, L.L.C. (2006)"
  holding="Courts must apply the traditional four-factor test before granting permanent injunctions in patent cases."
  rule="No presumption of injunctive relief exists; plaintiffs must prove irreparable harm, inadequacy of legal remedies, balance of hardships, and public interest."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/547/388",
    oyez: "https://www.oyez.org/cases/2005/05-130"
  }}
/>

<CaseCard
  title="Winter v. Natural Resources Defense Council, Inc. (2008)"
  holding="A preliminary injunction requires a clear showing of likely irreparable harm, not just a possibility, and courts must balance national security interests."
  rule="The four preliminary-injunction factors demand likelihood of success, irreparable harm, equities favoring the movant, and service of the public interest."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/07-1239",
    oyez: "https://www.oyez.org/cases/2008/07-1239"
  }}
/>

<CaseCard
  title="BMW of North America, Inc. v. Gore (1996)"
  holding="A $2 million punitive damages award for repainting a car violated due process because it was grossly excessive relative to the state's interests."
  rule="Punitive damages must be proportionate under guideposts examining reprehensibility, ratio to compensatory damages, and comparable civil penalties."
  links={{
    lii: "https://www.law.cornell.edu/supct/html/94-896.ZO.html",
    oyez: "https://www.oyez.org/cases/1995/94-896"
  }}
