Build doctrinal muscle memory by reviewing the Supreme Court opinions that anchor constitutional analysis.

<CaseCard
  title="Marbury v. Madison (1803)"
  holding="The Supreme Court may review executive actions and refuse to enforce laws that conflict with the Constitution."
  rule="Judicial review authorizes federal courts to invalidate statutes and executive acts that violate the Constitution, cementing the judiciary's role as interpreter of supreme law."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/5/137",
    oyez: "https://www.oyez.org/cases/1789-1850/5us137"
  }}
/>

<CaseCard
  title="McCulloch v. Maryland (1819)"
  holding="Congress may employ implied powers to carry out its enumerated responsibilities, and states cannot tax instruments of the national government."
  rule="The Necessary and Proper Clause authorizes means conducive to legitimate ends, while the Supremacy Clause forbids states from obstructing valid federal functions."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/17/316",
    oyez: "https://www.oyez.org/cases/1789-1850/17us316"
  }}
/>

<CaseCard
  title="Brown v. Board of Education (1954)"
  holding="State-sponsored segregation in public schools violates the Equal Protection Clause even when facilities are otherwise equal."
  rule="Laws separating students by race are inherently unequal and trigger strict scrutiny because they stigmatize minority groups and deny equal educational opportunities."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/347/483",
    oyez: "https://www.oyez.org/cases/1940-1955/347us483"
  }}
/>

<CaseCard
  title="United States v. Lopez (1995)"
  holding="Congress exceeded its Commerce Clause authority by criminalizing gun possession near schools without a substantial link to interstate commerce."
  rule="The Commerce Clause permits regulation of channels, instrumentalities, and activities substantially affecting commerce, but purely local non-economic conduct lies beyond federal reach absent a clear connection."
  links={{
    lii: "https://www.law.cornell.edu/supct/html/93-1260.ZO.html",
