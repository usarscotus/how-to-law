Revisit these decisions whenever you need quick doctrinal guardrails for the rules of proof.

<CaseCard
  title="Daubert v. Merrell Dow Pharmaceuticals, Inc. (1993)"
  holding="Federal judges must act as gatekeepers by ensuring that expert testimony rests on reliable principles and methods."
  rule="Rule 702 requires courts to assess scientific validity, including testing, peer review, error rates, and general acceptance, before admitting expert evidence."
  links={{
    lii: "https://www.law.cornell.edu/supct/html/92-102.ZS.html",

  }}
/>

<CaseCard
  title="Crawford v. Washington (2004)"
  holding="Admitting testimonial statements from an absent witness violates the Sixth Amendment unless the defendant had a prior opportunity for cross-examination."
  rule="The Confrontation Clause bars testimonial hearsay unless the declarant is unavailable and the accused had a chance to cross-examine."
  links={{
    lii: "https://www.law.cornell.edu/supct/html/02-9410.ZS.html",
    oyez: "https://www.oyez.org/cases/2003/02-9410"
  }}
/>

<CaseCard
  title="Huddleston v. United States (1988)"
  holding="Evidence of prior acts is admissible under Rule 404(b) if the jury could reasonably find by a preponderance that the acts occurred and the evidence is relevant for a non-propensity purpose."
  rule="Judges conduct a Rule 104(b) conditional relevance inquiry without weighing credibility, and must still balance probative value against unfair prejudice under Rule 403."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/485/681",
    oyez: "https://www.oyez.org/cases/1987/86-1757"
  }}
/>

<CaseCard
  title="Old Chief v. United States (1997)"
  holding="A court abuses its discretion by admitting the full record of a prior conviction when the defendant offers to stipulate to the status element and the record's unfair prejudice outweighs its probative value."
  rule="Rule 403 requires comparing evidentiary alternatives; when a stipulation satisfies an element with minimal prejudice, the prosecution must accept it."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/519/172",
    oyez: "https://www.oyez.org/cases/1996/95-6556"
  }}
/>
