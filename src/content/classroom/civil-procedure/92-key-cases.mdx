Use these classic opinions to orient your analysis whenever procedural puzzles arise.

<CaseCard
  title="International Shoe Co. v. Washington (1945)"
  holding="A state may exercise personal jurisdiction over an out-of-state defendant with minimum contacts such that the suit does not offend traditional notions of fair play and substantial justice."
  rule="Due process requires purposeful contacts and fairness considerations before a forum may bind a nonresident defendant."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/326/310",
    oyez: "https://www.oyez.org/cases/1940-1955/326us310"
  }}
/>

<CaseCard
  title="Erie Railroad Co. v. Tompkins (1938)"
  holding="Federal courts sitting in diversity must apply state substantive law rather than general federal common law."
  rule="The Erie doctrine ensures vertical uniformity by requiring federal courts to follow state substantive rules while applying federal procedural law."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/304/64",
    oyez: "https://www.oyez.org/cases/1900-1940/304us64"
  }}
/>

<CaseCard
  title="Hanna v. Plumer (1965)"
  holding="When a valid Federal Rule of Civil Procedure conflicts with state law in diversity cases, the federal rule governs."
  rule="The Rules Enabling Act authorizes application of procedural federal rules that regulate practice without abridging substantive rights, superseding contrary state procedure."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/380/460",
    oyez: "https://www.oyez.org/cases/1964/96"
  }}
/>

<CaseCard
  title="Celotex Corp. v. Catrett (1986)"
  holding="A party moving for summary judgment may point to the absence of evidence on an essential element, shifting the burden to the nonmovant to produce admissible proof."
  rule="Rule 56 allows judgment when no genuine dispute of material fact exists; after Celotex, movants may rely on evidentiary gaps to demonstrate entitlement to judgment as a matter of law."
  links={{
    lii: "https://www.law.cornell.edu/supremecourt/text/477/317",
    oyez: "https://www.oyez.org/cases/1985/85-198"
  }}
/>
